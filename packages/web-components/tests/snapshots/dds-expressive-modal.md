  id="start-sentinel"

  START
</button>

  aria-labelledby="modal-header"
  class="bx--modal-container"
  role="dialog"
  tabindex="-1"
>

    <div
      class="dds-ce--modal__hedaer--with-body"
      id="modal-header"
    >
      <slot name="header">
      </slot>
    </div>
    <div class="dds-ce--modal__body dds-ce--modal__body--with-footer">
      <slot>
      </slot>
    </div>
    <div>

      </slot>
    </div>
  </div>
</div>

  class="bx--visually-hidden"
  id="end-sentinel"
>
  END
</button>
