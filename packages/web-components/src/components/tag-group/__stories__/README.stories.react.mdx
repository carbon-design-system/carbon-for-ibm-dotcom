## Getting started

Here's a quick example to get you started.

### JS

```javascript
import C4DTagGroup from '@carbon/ibmdotcom-web-components/es/components-react/tag-group/tag-group.js';
import C4DTagLink from '@carbon/ibmdotcom-web-components/es/components-react/tag-link/tag-link.js';

function App() {
  return (
    <C4DTagGroup>
      <C4DTagLink href="https://example.com">Cloud</C4DTagLink>
      <C4DTagLink href="https://example.com">Blockchain</C4DTagLink>
      <C4DTagLink href="https://example.com">Supply chain</C4DTagLink>
      <C4DTagLink href="https://example.com">WebSphere</C4DTagLink>
    </C4DTagGroup>
  );
}
```
