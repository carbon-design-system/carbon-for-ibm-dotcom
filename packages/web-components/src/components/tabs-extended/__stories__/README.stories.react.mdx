## Getting started

Here's a quick example to get you started.

### JS

```javascript
import DDSTabsExtended from '@carbon/ibmdotcom-web-components/es/components-react/tabs-extended/tabs-extended.js';
import DDSTab from '@carbon/ibmdotcom-web-components/es/components-react/tabs-extended/tab.js';

function App() {
  return (
<DDSTabsExtended>
  <DDSTab label="First tab">
    <p>Content for first tab goes here.</p>
  </DDSTab>
  <DDSTab label="Second tab">
    <p>Content for second tab goes here.</p>
  </DDSTab>
</DDSTabsExtended>
  );
}
```
