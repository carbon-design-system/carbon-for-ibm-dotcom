## Getting started

Here's a quick example to get you started.

### JS

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import C4DCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import C4DCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import C4DCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import C4DCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';

function App() {
  return (
    <C4DCard href="https://example.com">
      <C4DCardEyebrow>Eyebrow text</C4DCardEyebrow>
      <C4DCardHeading>Lorem ipsum dolor sit amet</C4DCardHeading>
      <C4DCardFooter>
        <ArrowRight20 slot="icon" />
      </C4DCardFooter>
    </C4DCard>
  );
}
```

## Card with pictogram

You can browse the
[Carbon Pictogram library](https://www.carbondesignsystem.com/guidelines/pictograms/library/)
to find a pictogram that fits your design. You can also check out the repo
[here](https://github.com/carbon-design-system/carbon/tree/main/packages/pictograms)

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import C4DCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import C4DCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import Desktop from '@carbon/pictograms-react/lib/desktop/index.js';

function App() {
  return (
    <C4DCard pictogramPlacement="top" href="https://example.com">
      <C4DCardHeading>Lorem ipsum dolor sit amet</C4DCardHeading>
      <p>copy text</p>
      <Desktop slot="pictogram" />
    </C4DCard>
  );
}
```

## Card - static

When there is no URL being passed to the `C4DCard`'s href attribute, the static
variation will be rendered, where the card is no longer clickable and optional
`Tag` elements can also be added.

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import C4DCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import C4DCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import C4DCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import C4DCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';
import { Tag } from 'carbon-components-react';

function App() {
  return (
    <C4DCard colorScheme="light" border>
      <C4DCardEyebrow>Eyebrow text</C4DCardEyebrow>
      <C4DCardHeading>Heading text</C4DCardHeading>
      <div>
        <Tag>Most popular</Tag>
        <Tag type="purple">Enterprise</Tag>
      </div>
      <C4DCardFooter href="https://example.com">
        Card CTA text
        <ArrowRight20 slot="icon" />
      </C4DCardFooter>
    </C4DCard>
  );
}
```

## Card - link

```javascript
import C4DCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';

import C4DCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';

function App() {
  return (
    <C4DCard link href={href || undefined}>
      <C4DCardHeading>Card link heading </C4DCardHeading>
      <C4DCardFooter></C4DCardFooter>
    </C4DCard>
  );
}
```

## Card - logo

If you need to use a Logo image for the Card component, use the Logo variant to
ensure the Card properly accomodates to the Logo image.

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import C4DCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import C4DCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import C4DCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import C4DCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';
import { Tag } from 'carbon-components-react';

function App() {
  return (
    <C4DCard border logo href={href || undefined}>
      <C4DImageLogo
        slot="image"
        alt={alt}
        default-src={defaultSrc}></C4DImageLogo>
      {eyebrow ? <C4DCardEyebrow>{eyebrow}</C4DCardEyebrow> : ''}
      {heading ? <C4DCardHeading>{heading}</C4DCardHeading> : ''}
      {copy ? <p>{copy}</p> : ``}
      {tagGroup ? (
        <div>
          <Tag>Most popular</Tag>
          <Tag type="purple">Enterprise</Tag>
        </div>
      ) : (
        ''
      )}
      <C4DCardFooter></C4DCardFooter>
    </C4DCard>
  );
}
```

## Light theme

An optional light theme is available under `color-scheme`.

```javascript
<C4DCard colorScheme='light'>
```

## Tags

If using tags within `dds-card` in tandem with the dark themes, make sure the
proper styles are imported from `@carbon/styles` in your application.

```
@use '@carbon/styles/scss/themes' as *;
@use '@carbon/styles/scss/components/tag/tokens' as tag-tokens;
@include add-component-tokens(tag-tokens.$tag-tokens);
```

## Card with image

To add image to the `C4DCard`, import the `C4DImage` component and place
`C4DImage` within `C4DCard`:

```javascript
import C4DImage from '@carbon/ibmdotcom-web-components/es/components-react/image/image';
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import C4DCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import C4DCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import C4DCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import C4DCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';
import { Tag } from 'carbon-components-react';

function App() {
  return (
    <C4DCard colorScheme="light">
      <C4DImage slot="image" alt="alt text" defaultSrc="../image.png" />
      <C4DCardEyebrow>{eyebrow}</C4DCardEyebrow>
      <C4DCardHeading>{heading}</C4DCardHeading>
      <div>
        <Tag>Most popular</Tag>
        <Tag type="purple">Enterprise</Tag>
      </div>
      <C4DCardFooter href="https://www.example.com">
        Sign up for the trial
        <ArrowRight20 slot="icon" />
      </C4DCardFooter>
    </C4DCard>
  );
}
```

## Stable selectors

See
[our README](https://github.com/carbon-design-system/carbon-for-ibm-dotcom/tree/main/packages/web-components#stable-selectors-for-analytics-and-integratione2e-testing-in-web-components)

| Web Components selector | Compatibility selector            | Description |
| ----------------------- | --------------------------------- | ----------- |
| `<c4d-card>`            | `data-autoid="c4d--card"`         | Component   |
| `<c4d-card-eyebrow>`    | `data-autoid="c4d--card-eyebrow"` | Component   |
| `<c4d-card-heading>`    | `data-autoid="c4d--card-heading"` | Component   |
| `<c4d-card-footer>`     | `data-autoid="c4d--card-footer"`  | Component   |
