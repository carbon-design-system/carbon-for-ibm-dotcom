## Getting started

Here's a quick example to get you started.

### JS

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import DDSCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import DDSCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import DDSCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import DDSCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';

function App() {
  return (
    <DDSCard href="https://example.com">
      <DDSCardEyebrow>Eyebrow text</DDSCardEyebrow>
      <DDSCardHeading>Lorem ipsum dolor sit amet</DDSCardHeading>
      <DDSCardFooter>
        Card CTA text
        <ArrowRight20 slot="icon" />
      </DDSCardFooter>
    </DDSCard>
  );
}
```

## Card with pictogram

You can browse the [Carbon Pictogram library](https://www.carbondesignsystem.com/guidelines/pictograms/library/) to find a pictogram that fits your design.
You can also check out the repo [here](https://github.com/carbon-design-system/carbon/tree/main/packages/pictograms)

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import DDSCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import DDSCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import Desktop from '@carbon/pictograms-react/lib/desktop/index.js';

function App() {
  return (
    <DDSCard pictogramPlacement="top" href="https://example.com">
      <DDSCardHeading>Lorem ipsum dolor sit amet</DDSCardHeading>
      <p>copy text</p>
      <Desktop slot="pictogram" />
    </DDSCard>
  );
}
```

## Card - static

When there is no URL being passed to the `DDSCard`'s href attribute, the static variation will be rendered,
where the card is no longer clickable and optional `Tag` elements can also be added.

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import DDSCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import DDSCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import DDSCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import DDSCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';
import DDSTagGroup from '@carbon/ibmdotcom-web-components/es/components-react/tag-group/tag-group';
import { Tag } from 'carbon-components-react';

function App() {
  return (
    <DDSCard colorScheme="light" border>
      <DDSCardEyebrow>Eyebrow text</DDSCardEyebrow>
      <DDSCardHeading>Heading text</DDSCardHeading>
      <DDSTagGroup>
        <Tag>Most popular</Tag>
        <Tag type="purple">Enterprise</Tag>
      </DDSTagGroup>
      <DDSCardFooter href="https://example.com">
        Card CTA text
        <ArrowRight20 slot="icon" />
      </DDSCardFooter>
    </DDSCard>
  );
}
```

## Card - logo

If you need to use a Logo image for the Card component, use the Logo variant to ensure the Card
properly accomodates to the Logo image.

```javascript
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import DDSCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import DDSCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import DDSCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import DDSCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';
import DDSTagGroup from '@carbon/ibmdotcom-web-components/es/components-react/tag-group/tag-group';
import { Tag } from 'carbon-components-react';

function App() {
  return (
    <DDSCard border logo href={href || undefined}>
      <DDSImageLogo slot="image" alt={alt} default-src={defaultSrc}></DDSImageLogo>
      {eyebrow ? <DDSCardEyebrow>{eyebrow}</DDSCardEyebrow> : ''}
      {heading ? <DDSCardHeading>{heading}</DDSCardHeading> : ''}
      {copy ? <p>{copy}</p> : ``}
      {tagGroup ? (
        <DDSTagGroup>
          <Tag>Most popular</Tag>
          <Tag type="purple">Enterprise</Tag>
        </DDSTagGroup>
      ) : (
        ''
      )}
      <DDSCardFooter></DDSCardFooter>
    </DDSCard>
  );
}
```

## Light theme

```javascript
<DDSCard colorScheme='light'>
```

## Card with image

To add image to the `DDSCard`, import the `DDSImage` component and place `DDSImage` within `DDSCard`:

```javascript
import DDSImage from '@carbon/ibmdotcom-web-components/es/components-react/image/image';
import ArrowRight20 from '@carbon/icons-react/es/arrow--right/20.js';
import DDSCard from '@carbon/ibmdotcom-web-components/es/components-react/card/card';
import DDSCardHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-heading';
import DDSCardEyebrow from '@carbon/ibmdotcom-web-components/es/components-react/card/card-eyebrow';
import DDSCardFooter from '@carbon/ibmdotcom-web-components/es/components-react/card/card-footer';
import DDSTagGroup from '@carbon/ibmdotcom-web-components/es/components-react/tag-group/tag-group';
import { Tag } from 'carbon-components-react';

function App() {
  return (
    <DDSCard colorScheme="light">
      <DDSImage slot="image" alt="alt text" defaultSrc="../image.png" />
      <DDSCardEyebrow>{eyebrow}</DDSCardEyebrow>
      <DDSCardHeading>{heading}</DDSCardHeading>
      <DDSTagGroup>
        <Tag>Most popular</Tag>
        <Tag type="purple">Enterprise</Tag>
      </DDSTagGroup>
      <DDSCardFooter href="https://www.example.com">
        Sign up for the trial
        <ArrowRight20 slot="icon" />
      </DDSCardFooter>
    </DDSCard>
  );
}
```

## Stable selectors

See [our README](https://github.com/carbon-design-system/carbon-for-ibm-dotcom/tree/main/packages/web-components#stable-selectors-for-analytics-and-integratione2e-testing-in-web-components) to see how Web Components selector and `data-autoid` should be used.

| Web Components selector | Compatibility selector            | Description |
| ----------------------- | --------------------------------- | ----------- |
| `<dds-card>`            | `data-autoid="dds--card"`         | Component   |
| `<dds-card-eyebrow>`    | `data-autoid="dds--card-eyebrow"` | Component   |
| `<dds-card-heading>`    | `data-autoid="dds--card-heading"` | Component   |
| `<dds-card-footer>`     | `data-autoid="dds--card-footer"`  | Component   |
