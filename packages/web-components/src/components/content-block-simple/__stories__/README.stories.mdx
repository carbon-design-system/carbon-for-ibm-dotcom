## Getting started

Here's a quick example to get you started.

##### JS

```javascript
import '@carbon/ibmdotcom-web-components/es/components/image/image';
import '@carbon/ibmdotcom-web-components/es/components/cta/text-cta';
import '@carbon/ibmdotcom-web-components/es/components/content-block/content-block-heading';
import '@carbon/ibmdotcom-web-components/es/components/content-block-simple/content-block-simple';
```

### HTML

```html
<dds-content-block-simple .copy="Lorem ipsum *dolor* sit amet">
  <dds-content-block-heading>Curabitur malesuada varius mi eu posuere</dds-content-block-heading>
  <dds-image slot="media" alt="Image alt text" default-src="https://dummyimage.com/672x378/ee5396/161616&text=16:9">
    <dds-image-item media="(min-width: 672px)" srcset="https://dummyimage.com/672x378/ee5396/161616&text=16:9"> </dds-image-item>
    <dds-image-item media="(min-width: 400px)" srcset="https://dummyimage.com/400x225/ee5396/161616&text=16:9"> </dds-image-item>
    <dds-image-item media="(min-width: 320px)" srcset="https://dummyimage.com/320x180/ee5396/161616&text=16:9"> </dds-image-item>
  </dds-image>
  <dds-text-cta slot="cta" href="https://www.ibm.com/" type="local">Lorem ipsum dolor sit amet</dds-text-cta>
</dds-content-block-simple>
```
