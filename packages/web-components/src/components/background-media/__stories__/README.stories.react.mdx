## Getting started

Here's a quick example to get you started.

### JS

#### With Background Image

```javascript
import C4DBackgroundMedia from '@carbon/ibmdotcom-web-components/es/components-react/background-media/background-media';
import C4DImageItem from '@carbon/ibmdotcom-web-components/es/components-react/image/image-item';

function App() {
  return (
    <C4DBackgroundMedia
      gradient-direction="left-to-right"
      mobile-position=""
      alt="Image alt text"
      default-src="https://fpoimg.com/1584x560?&bg_color=ee5396&text_color=161616">
      <C4DImageItem
        media="(min-width: 1584px)"
        srcset="https://fpoimg.com/1584x560?&bg_color=ee5396&text_color=161616">
        {' '}
      </C4DImageItem>
      <C4DImageItem
        media="(min-width: 1312px)"
        srcset="https://fpoimg.com/1594x887?text=16:9&bg_color=ee5396&text_color=161616">
        {' '}
      </C4DImageItem>
      <C4DImageItem
        media="(min-width: 672px)"
        srcset="https://fpoimg.com/1594x887?text=16:9&bg_color=ee5396&text_color=161616">
        {' '}
      </C4DImageItem>
      <C4DImageItem
        media="(min-width: 320px)"
        srcset="https://fpoimg.com/480x360?text=4:3&bg_color=ee5396&text_color=161616">
        {' '}
      </C4DImageItem>
      <C4DImageItem
        media="(min-width: 0px)"
        srcset="https://fpoimg.com/480x360?text=4:3&bg_color=ee5396&text_color=161616">
        {' '}
      </C4DImageItem>
    </C4DBackgroundMedia>
  );
}
```

#### With Background Video

```javascript
import C4DBackgroundMedia from '@carbon/ibmdotcom-web-components/es/components-react/background-media/background-media';
import C4DVideoPlayerContainer from '@carbon/ibmdotcom-web-components/es/components-react/video-player/video-player-container';

function App() {
  return (
    <C4DBackgroundMedia
      gradient-direction="left-to-right"
      mobile-position=""
      alt="Image alt text"
      default-src="https://fpoimg.com/1584x560?&bg_color=ee5396&text_color=161616">
      <C4DVideoPlayerContainer
        video-id="0_ibuqxqbe"
        background-mode="true"></C4DVideoPlayerContainer>
    </C4DBackgroundMedia>
  );
}
```
