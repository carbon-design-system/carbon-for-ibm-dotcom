## Getting started

Here's a quick example to get you started.

### JS

```javascript
import '@carbon/ibmdotcom-web-components/es/components/leadspace/leadspace.js';
import '@carbon/ibmdotcom-web-components/es/components/image/image.js';
import ArrowRight20 from 'carbon-web-components/es/icons/arrow--right/20.js';
import '@carbon/ibmdotcom-web-components/es/components/button-group/button-group.js';
import '@carbon/ibmdotcom-web-components/es/components/button-group/button-group-item.js';
```

### HTML

```html
<dds-leadspace alt="alt text" default-src="www.example.com">
  <span slot="title">LeadSpace title</span>
  <span slot="copy">LeadSpace copy</span>
  <dds-button-group slot="buttons">
    ${buttons.map( elem => html`
    <dds-button-group-item href="www.example.com">${elem.copy}${ArrowRight20}</dds-button-group-item>
    ` )}
  </dds-button-group>
  <dds-image slot="image" class="bx--image" alt="alt text" default-src="https://picsum.photos/id/1076/1056/480">
    <dds-image-item media="(min-width: 672px)" srcset="https://picsum.photos/id/1076/672/400"></dds-image-item>
    <dds-image-item media="(min-width: 0)" srcset="https://picsum.photos/id/1076/320/370"></dds-image-item>
  </dds-image>
</dds-leadspace>
```
