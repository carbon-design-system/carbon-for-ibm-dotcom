### JS (with Card Link)

```javascript
import C4DCardGroup from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group.js';
import C4DCardGroupCardLinkItem from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group-card-link-item.js';
import C4DCardLinkHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-link-heading.js';
import C4DCardCTAFooter from '@carbon/ibmdotcom-web-components/es/components-react/cta/card-cta-footer.js';

function App() {
  return (
    <C4DCardGroup gridMode="narrow">
      <C4DCardGroupCardLinkItem
        cta-type="local"
        href="https://example.com"
        pattern-background>
        <C4DCardLinkHeading>IBM Developer - Card 01</C4DCardLinkHeading>
        <p>Learn, code and connect with your community</p>
        <C4DCardCTAFooter slot="footer"></C4DCardCTAFooter>
      </C4DCardGroupCardLinkItem>
    </C4DCardGroup>
  );
}
```

### JS (with Mixed Media)

In order to have video components, make sure to import the
`C4DVideoCTAContainer` and place `C4DCardGroup` within.

To set the `C4DCardInCard` to play video, see more details
[here](https://www.ibm.com/standards/carbon/web-components/react/?path=/docs/components-card-in-card--default).

```javascript
import C4DVideoCTAContainer from '@carbon/ibmdotcom-web-components/es/components-react/cta/video-cta-container';
import C4DCardGroup from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group.js';
import C4DCardGroupItem from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group-item.js';
import C4DCardLinkHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-link-heading.js';
import C4DCardCTAFooter from '@carbon/ibmdotcom-web-components/es/components-react/cta/card-cta-footer.js';

function App() {
  return (
    <C4DVideoCTAContainer>
      <C4DCardGroup gridMode="narrow">
        <C4DCardGroupItem cta-type="video" href="0_ibuqxqbe">
          <C4DCardEyebrow>Topic</C4DCardEyebrow>
          <C4DCardCTAFooter
            cta-type="video"
            href="0_ibuqxqbe"
            slot="footer"></C4DCardCTAFooter>
        </C4DCardGroupItem>
      </C4DCardGroup>
    </C4DVideoCTAContainer>
  );
}
```
