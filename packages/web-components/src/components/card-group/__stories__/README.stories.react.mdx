### JS (with Card Link)

```javascript
import DDSCardGroup from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group.js';
import DDSCardGroupCardLinkItem from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group-card-link-item.js';
import DDSCardLinkHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-link-heading.js';
import DDSCardCTAFooter from '@carbon/ibmdotcom-web-components/es/components-react/cta/card-cta-footer.js';

function App() {
  return (
    <DDSCardGroup gridMode="narrow">
      <DDSCardGroupCardLinkItem cta-type="local" href="https://example.com" pattern-background>
        <DDSCardLinkHeading>IBM Developer - Card 01</DDSCardLinkHeading>
        <p>Learn, code and connect with your community</p>
        <DDSCardCTAFooter slot="footer"></DDSCardCTAFooter>
      </DDSCardGroupCardLinkItem>
    </DDSCardGroup>
  );
}
```

### JS (with Mixed Media)

In order to have video components, make sure to import the `DDSVideoCTAContainer` and place `DDSCardGroup` within.

To set the `DDSCardInCard` to play video, see more details [here](https://www.ibm.com/standards/carbon/web-components/react/?path=/docs/components-card-in-card--default).

```javascript
import DDSVideoCTAContainer from '@carbon/ibmdotcom-web-components/es/components-react/cta/video-cta-container';
import DDSCardGroup from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group.js';
import DDSCardGroupItem from '@carbon/ibmdotcom-web-components/es/components-react/card-group/card-group-item.js';
import DDSCardLinkHeading from '@carbon/ibmdotcom-web-components/es/components-react/card/card-link-heading.js';
import DDSCardCTAFooter from '@carbon/ibmdotcom-web-components/es/components-react/cta/card-cta-footer.js';

function App() {
  return (
    <DDSVideoCTAContainer>
      <DDSCardGroup gridMode="narrow">
        <DDSCardGroupItem cta-type="video" href="1_9h94wo6b">
          <DDSCardEyebrow>Topic</DDSCardEyebrow>
          <DDSCardCTAFooter cta-type="video" href="1_9h94wo6b" slot="footer"></DDSCardCTAFooter>
        </DDSCardGroupItem>
      </DDSCardGroup>
    </DDSVideoCTAContainer>
  );
}
```
