## Getting started

Here's a quick example to get you started.

### JS (with images)

```javascript
import C4DExpressiveModal from '@carbon/ibmdotcom-web-components/es/components-react/expressive-modal/expressive-modal';
import C4DExpressiveModalCloseButton from '@carbon/ibmdotcom-web-components/es/components-react/expressive-modal/expressive-modal-close-button';
import C4DLightboxMediaViewer from '@carbon/ibmdotcom-web-components/es/components-react/lightbox-media-viewer/lightbox-media-viewer';

function App() {
  return (
    <C4DExpressiveModal id="my-lightbox" expressive-size="full-width">
      <C4DExpressiveModalCloseButton></C4DExpressiveModalCloseButton>
      <C4DLightboxMediaViewer
        alt="Image alt text"
        default-src="https://fpoimg.com/1280x720?text=16:9&bg_color=ee5396&text_color=161616"
        description="Lorem ipsum dolor sit amet, consectetur adipiscing elit. Aenean et ultricies est.Mauris iaculis eget dolor nec hendrerit."
        title="Curabitur malesuada varius mi eu posuere"></C4DLightboxMediaViewer>
    </C4DExpressiveModal>
  );
}
```

### JS (with videos)

```javascript
import C4DExpressiveModal from '@carbon/ibmdotcom-web-components/es/components-react/expressive-modal/expressive-modal';
import C4DExpressiveModalCloseButton from '@carbon/ibmdotcom-web-components/es/components-react/expressive-modal/expressive-modal-close-button';
import C4DLightboxMediaViewer from '@carbon/ibmdotcom-web-components/es/components-react/lightbox-media-viewer/lightbox-media-viewer';

function App() {
  return (
    <C4DExpressiveModal id="my-lightbox" expressive-size="full-width">
      <C4DExpressiveModalCloseButton></C4DExpressiveModalCloseButton>
      <C4DLightboxMediaViewer
        id="my-video"
        video-id="0_ibuqxqbe"></C4DLightboxMediaViewer>
    </C4DExpressiveModal>
  );
}
```

### JS (multiple images/videos)

```javascript
import C4DExpressiveModal from '@carbon/ibmdotcom-web-components/es/components-react/expressive-modal/expressive-modal';

function App() {
  return (
  <C4DExpressiveModal id="my-lightbox" expressive-size="full-width">
    <C4DExpressiveModalCloseButton></C4DExpressiveModalCloseButton>
    <C4DCarousel>
      <C4DLightboxMediaViewer ... ></C4DLightboxMediaViewer>
      <C4DLightboxMediaViewer ... ></C4DLightboxMediaViewer>

      <C4DLightboxMediaViewer ... ></C4DLightboxMediaViewer>
    </C4DCarousel>
  </C4DExpressiveModal>
  );
}
```
